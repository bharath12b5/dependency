213     /**
214      * Loads a specified JAR file and collects information from the manifest and checksums to identify the correct CPE
215      * information.
216      *
217      * @param dependency the dependency to analyze.
218      * @param engine the engine that is scanning the dependencies
219      * @throws AnalysisException is thrown if there is an error reading the JAR file.
220      */
221     @Override
222     public void analyzeFileType(Dependency dependency, Engine engine) throws AnalysisException {
223         try {
224             final List<ClassNameInformation> classNames = collectClassNames(dependency);
225             final String fileName = dependency.getFileName().toLowerCase();
226             if (classNames.isEmpty()
227                     && (fileName.endsWith("-sources.jar")
228                     || fileName.endsWith("-javadoc.jar")
229                     || fileName.endsWith("-src.jar")
230                     || fileName.endsWith("-doc.jar"))) {
231                 engine.getDependencies().remove(dependency);
232             }
233             final boolean hasManifest = parseManifest(dependency, classNames);
234             final boolean hasPOM = analyzePOM(dependency, classNames, engine);
235             final boolean addPackagesAsEvidence = !(hasManifest && hasPOM);
236             analyzePackageNames(classNames, dependency, addPackagesAsEvidence);
237         } catch (IOException ex) {
238             throw new AnalysisException("Exception occurred reading the JAR file.", ex);
239         }
240     }

242     /**
243      * Attempts to find a pom.xml within the JAR file. If found it extracts information and adds it to the evidence. This will
244      * attempt to interpolate the strings contained within the pom.properties if one exists.
245      *
246      * @param dependency the dependency being analyzed
247      * @param classes a collection of class name information
248      * @param engine the analysis engine, used to add additional dependencies
249      * @throws AnalysisException is thrown if there is an exception parsing the pom
250      * @return whether or not evidence was added to the dependency
251      */
252     protected boolean analyzePOM(Dependency dependency, List<ClassNameInformation> classes, Engine engine) throws AnalysisException {
253         boolean foundSomething = false;
254         final JarFile jar;
255         try {
256             jar = new JarFile(dependency.getActualFilePath());
257         } catch (IOException ex) {
258             final String msg = String.format("Unable to read JarFile '%s'.", dependency.getActualFilePath());
259             //final AnalysisException ax = new AnalysisException(msg, ex);
260             LOGGER.log(Level.WARNING, msg);
261             LOGGER.log(Level.FINE, "", ex);
262             return false;
263         }
264         List<String> pomEntries;
265         try {
266             pomEntries = retrievePomListing(jar);
267         } catch (IOException ex) {
268             final String msg = String.format("Unable to read Jar file entries in '%s'.", dependency.getActualFilePath());
269             //final AnalysisException ax = new AnalysisException(msg, ex);
270             LOGGER.log(Level.WARNING, msg);
271             LOGGER.log(Level.FINE, msg, ex);
272             return false;
273         }
274         File externalPom = null;
275         if (pomEntries.isEmpty()) {
276             String pomPath = dependency.getActualFilePath();
277             pomPath = pomPath.substring(0, pomPath.lastIndexOf('.')) + ".pom";
278             externalPom = new File(pomPath);
279             if (externalPom.isFile()) {
280                 pomEntries.add(pomPath);
281             } else {
282                 return false;
283             }
284         }
285         for (String path : pomEntries) {
286             LOGGER.fine(String.format("Reading pom entry: %s", path));
287             Properties pomProperties = null;
288             try {
289                 if (externalPom == null) {
290                     pomProperties = retrievePomProperties(path, jar);
291                 }
292             } catch (IOException ex) {
293                 LOGGER.log(Level.FINEST, "ignore this, failed reading a non-existent pom.properties", ex);
294             }
295             Model pom = null;
296             try {
297                 if (pomEntries.size() > 1) {
298                     //extract POM to its own directory and add it as its own dependency
299                     final Dependency newDependency = new Dependency();
300                     pom = extractPom(path, jar, newDependency);
301 
302                     final String displayPath = String.format("%s%s%s",
303                             dependency.getFilePath(),
304                             File.separator,
305                             path);
306                     final String displayName = String.format("%s%s%s",
307                             dependency.getFileName(),
308                             File.separator,
309                             path);
310 
311                     newDependency.setFileName(displayName);
312                     newDependency.setFilePath(displayPath);
313                     setPomEvidence(newDependency, pom, pomProperties, null);
314                     engine.getDependencies().add(newDependency);
315                     Collections.sort(engine.getDependencies());
316                 } else {
317                     if (externalPom == null) {
318                         pom = PomUtils.readPom(path, jar);
319                     } else {
320                         pom = PomUtils.readPom(externalPom);
321                     }
322                     foundSomething |= setPomEvidence(dependency, pom, pomProperties, classes);
323                 }
324             } catch (AnalysisException ex) {
325                 final String msg = String.format("An error occured while analyzing '%s'.", dependency.getActualFilePath());
326                 LOGGER.log(Level.WARNING, msg);
327                 LOGGER.log(Level.FINE, "", ex);
328             }
329         }
330         return foundSomething;
331     }

333     /**
334      * Given a path to a pom.xml within a JarFile, this method attempts to load a sibling pom.properties if one exists.
335      *
336      * @param path the path to the pom.xml within the JarFile
337      * @param jar the JarFile to load the pom.properties from
338      * @return a Properties object or null if no pom.properties was found
339      * @throws IOException thrown if there is an exception reading the pom.properties
340      */
341     private Properties retrievePomProperties(String path, final JarFile jar) throws IOException {
342         Properties pomProperties = null;
343         final String propPath = path.substring(0, path.length() - 7) + "pom.properies";
344         final ZipEntry propEntry = jar.getEntry(propPath);
345         if (propEntry != null) {
346             Reader reader = null;
347             try {
348                 reader = new InputStreamReader(jar.getInputStream(propEntry), "UTF-8");
349                 pomProperties = new Properties();
350                 pomProperties.load(reader);
351                 LOGGER.fine(String.format("Read pom.properties: %s", propPath));
352             } finally {
353                 if (reader != null) {
354                     try {
355                         reader.close();
356                     } catch (IOException ex) {
357                         LOGGER.log(Level.FINEST, "close error", ex);
358                     }
359                 }
360             }
361         }
362         return pomProperties;
363     }

365     /**
366      * Searches a JarFile for pom.xml entries and returns a listing of these entries.
367      *
368      * @param jar the JarFile to search
369      * @return a list of pom.xml entries
370      * @throws IOException thrown if there is an exception reading a JarEntry
371      */
372     private List<String> retrievePomListing(final JarFile jar) throws IOException {
373         final List<String> pomEntries = new ArrayList<String>();
374         final Enumeration<JarEntry> entries = jar.entries();
375         while (entries.hasMoreElements()) {
376             final JarEntry entry = entries.nextElement();
377             final String entryName = (new File(entry.getName())).getName().toLowerCase();
378             if (!entry.isDirectory() && "pom.xml".equals(entryName)) {
379                 LOGGER.fine(String.format("POM Entry found: %s", entry.getName()));
380                 pomEntries.add(entry.getName());
381             }
382         }
383         return pomEntries;
384     }

386     /**
387      * Retrieves the specified POM from a jar file and converts it to a Model.
388      *
389      * @param path the path to the pom.xml file within the jar file
390      * @param jar the jar file to extract the pom from
391      * @param dependency the dependency being analyzed
392      * @return returns the POM object
393      * @throws AnalysisException is thrown if there is an exception extracting or parsing the POM
394      * {@link org.owasp.dependencycheck.jaxb.pom.generated.Model} object
395      */
396     private Model extractPom(String path, JarFile jar, Dependency dependency) throws AnalysisException {
397         InputStream input = null;
398         FileOutputStream fos = null;
399         BufferedOutputStream bos = null;
400         final File tmpDir = getNextTempDirectory();
401         final File file = new File(tmpDir, "pom.xml");
402         try {
403             final ZipEntry entry = jar.getEntry(path);
404             input = jar.getInputStream(entry);
405             fos = new FileOutputStream(file);
406             bos = new BufferedOutputStream(fos, BUFFER_SIZE);
407             int count;
408             final byte[] data = new byte[BUFFER_SIZE];
409             while ((count = input.read(data, 0, BUFFER_SIZE)) != -1) {
410                 bos.write(data, 0, count);
411             }
412             bos.flush();
413             dependency.setActualFilePath(file.getAbsolutePath());
414         } catch (IOException ex) {
415             final String msg = String.format("An error occurred reading '%s' from '%s'.", path, dependency.getFilePath());
416             LOGGER.warning(msg);
417             LOGGER.log(Level.SEVERE, "", ex);
418         } finally {
419             closeStream(bos);
420             closeStream(fos);
421             closeStream(input);
422         }
423         Model model = PomUtils.readPom(file);
424         return model;
425     }

457     /**
458      * Sets evidence from the pom on the supplied dependency.
459      *
460      * @param dependency the dependency to set data on
461      * @param pom the information from the pom
462      * @param pomProperties the pom properties file (null if none exists)
463      * @param classes a collection of ClassNameInformation - containing data about the fully qualified class names within the JAR
464      * file being analyzed
465      * @return true if there was evidence within the pom that we could use; otherwise false
466      */
467     private boolean setPomEvidence(Dependency dependency, Model pom, Properties pomProperties, List<ClassNameInformation> classes) {
468         boolean foundSomething = false;
469         boolean addAsIdentifier = true;
470         if (pom == null) {
471             return foundSomething;
472         }
473         String groupid = interpolateString(pom.getGroupId(), pomProperties);
474         String parentGroupId = null;
475 
476         if (pom.getParentGroupId() != null) {
477             parentGroupId = interpolateString(pom.getParentGroupId(), pomProperties);
478             if ((groupid == null || groupid.isEmpty()) && parentGroupId != null && !parentGroupId.isEmpty()) {
479                 groupid = parentGroupId;
480             }
481         }
482         final String originalGroupID = groupid;
483 
484         if (groupid != null && !groupid.isEmpty()) {
485             foundSomething = true;
486             dependency.getVendorEvidence().addEvidence("pom", "groupid", groupid, Confidence.HIGHEST);
487             dependency.getProductEvidence().addEvidence("pom", "groupid", groupid, Confidence.LOW);
488             addMatchingValues(classes, groupid, dependency.getVendorEvidence());
489             addMatchingValues(classes, groupid, dependency.getProductEvidence());
490             if (parentGroupId != null && !parentGroupId.isEmpty() && !parentGroupId.equals(groupid)) {
491                 dependency.getVendorEvidence().addEvidence("pom", "parent-groupid", parentGroupId, Confidence.MEDIUM);
492                 dependency.getProductEvidence().addEvidence("pom", "parent-groupid", parentGroupId, Confidence.LOW);
493                 addMatchingValues(classes, parentGroupId, dependency.getVendorEvidence());
494                 addMatchingValues(classes, parentGroupId, dependency.getProductEvidence());
495             }
496         } else {
497             addAsIdentifier = false;
498         }
499 
500         String artifactid = interpolateString(pom.getArtifactId(), pomProperties);
501         String parentArtifactId = null;
502 
503         if (pom.getParentArtifactId() != null) {
504             parentArtifactId = interpolateString(pom.getParentArtifactId(), pomProperties);
505             if ((artifactid == null || artifactid.isEmpty()) && parentArtifactId != null && !parentArtifactId.isEmpty()) {
506                 artifactid = parentArtifactId;
507             }
508         }
509         final String originalArtifactID = artifactid;
510         if (artifactid != null && !artifactid.isEmpty()) {
511             if (artifactid.startsWith("org.") || artifactid.startsWith("com.")) {
512                 artifactid = artifactid.substring(4);
513             }
514             foundSomething = true;
515             dependency.getProductEvidence().addEvidence("pom", "artifactid", artifactid, Confidence.HIGHEST);
516             dependency.getVendorEvidence().addEvidence("pom", "artifactid", artifactid, Confidence.LOW);
517             addMatchingValues(classes, artifactid, dependency.getVendorEvidence());
518             addMatchingValues(classes, artifactid, dependency.getProductEvidence());
519             if (parentArtifactId != null && !parentArtifactId.isEmpty() && !parentArtifactId.equals(artifactid)) {
520                 dependency.getProductEvidence().addEvidence("pom", "parent-artifactid", parentArtifactId, Confidence.MEDIUM);
521                 dependency.getVendorEvidence().addEvidence("pom", "parent-artifactid", parentArtifactId, Confidence.LOW);
522                 addMatchingValues(classes, parentArtifactId, dependency.getVendorEvidence());
523                 addMatchingValues(classes, parentArtifactId, dependency.getProductEvidence());
524             }
525         } else {
526             addAsIdentifier = false;
527         }
528         //version
529         String version = interpolateString(pom.getVersion(), pomProperties);
530         String parentVersion = null;
531 
532         if (pom.getParentVersion() != null) {
533             parentVersion = interpolateString(pom.getParentVersion(), pomProperties);
534             if ((version == null || version.isEmpty()) && parentVersion != null && !parentVersion.isEmpty()) {
535                 version = parentVersion;
536             }
537         }
538 
539         if (version != null && !version.isEmpty()) {
540             foundSomething = true;
541             dependency.getVersionEvidence().addEvidence("pom", "version", version, Confidence.HIGHEST);
542             if (parentVersion != null && !parentVersion.isEmpty() && !parentVersion.equals(version)) {
543                 dependency.getVersionEvidence().addEvidence("pom", "parent-version", version, Confidence.LOW);
544             }
545         } else {
546             addAsIdentifier = false;
547         }
548 
549         if (addAsIdentifier) {
550             dependency.addIdentifier("maven", String.format("%s:%s:%s", originalGroupID, originalArtifactID, version), null, Confidence.LOW);
551         }
552 
553         // org name
554         final String org = pom.getOrganization();
555         if (org != null) {
556             foundSomething = true;
557             final String orgName = interpolateString(org, pomProperties);
558             if (orgName != null && !orgName.isEmpty()) {
559                 dependency.getVendorEvidence().addEvidence("pom", "organization name", orgName, Confidence.HIGH);
560                 addMatchingValues(classes, orgName, dependency.getVendorEvidence());
561             }
562         }
563         //pom name
564         final String pomName = interpolateString(pom.getName(), pomProperties);
565         if (pomName != null && !pomName.isEmpty()) {
566             foundSomething = true;
567             dependency.getProductEvidence().addEvidence("pom", "name", pomName, Confidence.HIGH);
568             dependency.getVendorEvidence().addEvidence("pom", "name", pomName, Confidence.HIGH);
569             addMatchingValues(classes, pomName, dependency.getVendorEvidence());
570             addMatchingValues(classes, pomName, dependency.getProductEvidence());
571         }
572 
573         //Description
574         if (pom.getDescription() != null) {
575             foundSomething = true;
576             final String description = interpolateString(pom.getDescription(), pomProperties);
577             if (description != null && !description.isEmpty()) {
578                 final String trimmedDescription = addDescription(dependency, description, "pom", "description");
579                 addMatchingValues(classes, trimmedDescription, dependency.getVendorEvidence());
580                 addMatchingValues(classes, trimmedDescription, dependency.getProductEvidence());
581             }
582         }
583         extractLicense(pom, pomProperties, dependency);
584         return foundSomething;
585     }

587     /**
588      * Analyzes the path information of the classes contained within the JarAnalyzer to try and determine possible vendor or
589      * product names. If any are found they are stored in the packageVendor and packageProduct hashSets.
590      *
591      * @param classNames a list of class names
592      * @param dependency a dependency to analyze
593      * @param addPackagesAsEvidence a flag indicating whether or not package names should be added as evidence.
594      */
595     protected void analyzePackageNames(List<ClassNameInformation> classNames,
596             Dependency dependency, boolean addPackagesAsEvidence) {
597         final Map<String, Integer> vendorIdentifiers = new HashMap<String, Integer>();
598         final Map<String, Integer> productIdentifiers = new HashMap<String, Integer>();
599         analyzeFullyQualifiedClassNames(classNames, vendorIdentifiers, productIdentifiers);
600 
601         final int classCount = classNames.size();
602         final EvidenceCollection vendor = dependency.getVendorEvidence();
603         final EvidenceCollection product = dependency.getProductEvidence();
604 
605         for (Map.Entry<String, Integer> entry : vendorIdentifiers.entrySet()) {
606             final float ratio = entry.getValue() / (float) classCount;
607             if (ratio > 0.5) {
608                 //TODO remove weighting
609                 vendor.addWeighting(entry.getKey());
610                 if (addPackagesAsEvidence && entry.getKey().length() > 1) {
611                     vendor.addEvidence("jar", "package name", entry.getKey(), Confidence.LOW);
612                 }
613             }
614         }
615         for (Map.Entry<String, Integer> entry : productIdentifiers.entrySet()) {
616             final float ratio = entry.getValue() / (float) classCount;
617             if (ratio > 0.5) {
618                 product.addWeighting(entry.getKey());
619                 if (addPackagesAsEvidence && entry.getKey().length() > 1) {
620                     product.addEvidence("jar", "package name", entry.getKey(), Confidence.LOW);
621                 }
622             }
623         }
624     }

626     /**
627      * <p>
628      * Reads the manifest from the JAR file and collects the entries. Some vendorKey entries are:</p>
629      * <ul><li>Implementation Title</li>
630      * <li>Implementation Version</li> <li>Implementation Vendor</li>
631      * <li>Implementation VendorId</li> <li>Bundle Name</li> <li>Bundle Version</li> <li>Bundle Vendor</li> <li>Bundle
632      * Description</li> <li>Main Class</li> </ul>
633      * However, all but a handful of specific entries are read in.
634      *
635      * @param dependency A reference to the dependency
636      * @param classInformation a collection of class information
637      * @return whether evidence was identified parsing the manifest
638      * @throws IOException if there is an issue reading the JAR file
639      */
640     protected boolean parseManifest(Dependency dependency, List<ClassNameInformation> classInformation) throws IOException {
641         boolean foundSomething = false;
642         JarFile jar = null;
643         try {
644             jar = new JarFile(dependency.getActualFilePath());
645 
646             final Manifest manifest = jar.getManifest();
647 
648             if (manifest == null) {
649                 //don't log this for javadoc or sources jar files
650                 if (!dependency.getFileName().toLowerCase().endsWith("-sources.jar")
651                         && !dependency.getFileName().toLowerCase().endsWith("-javadoc.jar")
652                         && !dependency.getFileName().toLowerCase().endsWith("-src.jar")
653                         && !dependency.getFileName().toLowerCase().endsWith("-doc.jar")) {
654                     LOGGER.log(Level.FINE,
655                             String.format("Jar file '%s' does not contain a manifest.",
656                                     dependency.getFileName()));
657                 }
658                 return false;
659             }
660             final Attributes atts = manifest.getMainAttributes();
661 
662             final EvidenceCollection vendorEvidence = dependency.getVendorEvidence();
663             final EvidenceCollection productEvidence = dependency.getProductEvidence();
664             final EvidenceCollection versionEvidence = dependency.getVersionEvidence();
665 
666             final String source = "Manifest";
667 
668             for (Entry<Object, Object> entry : atts.entrySet()) {
669                 String key = entry.getKey().toString();
670                 String value = atts.getValue(key);
671                 if (HTML_DETECTION_PATTERN.matcher(value).find()) {
672                     value = Jsoup.parse(value).text();
673                 }
674                 if (IGNORE_VALUES.contains(value)) {
675                     continue;
676                 } else if (key.equalsIgnoreCase(Attributes.Name.IMPLEMENTATION_TITLE.toString())) {
677                     foundSomething = true;
678                     productEvidence.addEvidence(source, key, value, Confidence.HIGH);
679                     addMatchingValues(classInformation, value, productEvidence);
680                 } else if (key.equalsIgnoreCase(Attributes.Name.IMPLEMENTATION_VERSION.toString())) {
681                     foundSomething = true;
682                     versionEvidence.addEvidence(source, key, value, Confidence.HIGH);
683                 } else if (key.equalsIgnoreCase(Attributes.Name.IMPLEMENTATION_VENDOR.toString())) {
684                     foundSomething = true;
685                     vendorEvidence.addEvidence(source, key, value, Confidence.HIGH);
686                     addMatchingValues(classInformation, value, vendorEvidence);
687                 } else if (key.equalsIgnoreCase(Attributes.Name.IMPLEMENTATION_VENDOR_ID.toString())) {
688                     foundSomething = true;
689                     vendorEvidence.addEvidence(source, key, value, Confidence.MEDIUM);
690                     addMatchingValues(classInformation, value, vendorEvidence);
691                 } else if (key.equalsIgnoreCase(BUNDLE_DESCRIPTION)) {
692                     foundSomething = true;
693                     addDescription(dependency, value, "manifest", key);
694                     //productEvidence.addEvidence(source, key, value, Confidence.MEDIUM);
695                     addMatchingValues(classInformation, value, productEvidence);
696                 } else if (key.equalsIgnoreCase(BUNDLE_NAME)) {
697                     foundSomething = true;
698                     productEvidence.addEvidence(source, key, value, Confidence.MEDIUM);
699                     addMatchingValues(classInformation, value, productEvidence);
700                 } else if (key.equalsIgnoreCase(BUNDLE_VENDOR)) {
701                     foundSomething = true;
702                     vendorEvidence.addEvidence(source, key, value, Confidence.HIGH);
703                     addMatchingValues(classInformation, value, vendorEvidence);
704                 } else if (key.equalsIgnoreCase(BUNDLE_VERSION)) {
705                     foundSomething = true;
706                     versionEvidence.addEvidence(source, key, value, Confidence.HIGH);
707                 } else if (key.equalsIgnoreCase(Attributes.Name.MAIN_CLASS.toString())) {
708                     continue;
709                     //skipping main class as if this has important information to add
710                     // it will be added during class name analysis...  if other fields
711                     // have the information from the class name then they will get added...
712 //                    foundSomething = true;
713 //                    productEvidence.addEvidence(source, key, value, Confidence.MEDIUM);
714 //                    vendorEvidence.addEvidence(source, key, value, Confidence.MEDIUM);
715 //                    addMatchingValues(classInformation, value, vendorEvidence);
716 //                    addMatchingValues(classInformation, value, productEvidence);
717                 } else {
718                     key = key.toLowerCase();
719 
720                     if (!IGNORE_KEYS.contains(key)
721                             && !key.endsWith("jdk")
722                             && !key.contains("lastmodified")
723                             && !key.endsWith("package")
724                             && !key.endsWith("classpath")
725                             && !key.endsWith("class-path")
726                             && !key.endsWith("-scm") //todo change this to a regex?
727                             && !key.startsWith("scm-")
728                             && !value.trim().startsWith("scm:")
729                             && !isImportPackage(key, value)
730                             && !isPackage(key, value)) {
731 
732                         foundSomething = true;
733                         if (key.contains("version")) {
734                             if (key.contains("specification")) {
735                                 versionEvidence.addEvidence(source, key, value, Confidence.LOW);
736                             } else {
737                                 versionEvidence.addEvidence(source, key, value, Confidence.MEDIUM);
738                             }
739                         } else if ("build-id".equals(key)) {
740                             int pos = value.indexOf('(');
741                             if (pos >= 0) {
742                                 value = value.substring(0, pos - 1);
743                             }
744                             pos = value.indexOf('[');
745                             if (pos >= 0) {
746                                 value = value.substring(0, pos - 1);
747                             }
748                             versionEvidence.addEvidence(source, key, value, Confidence.MEDIUM);
749                         } else if (key.contains("title")) {
750                             productEvidence.addEvidence(source, key, value, Confidence.MEDIUM);
751                             addMatchingValues(classInformation, value, productEvidence);
752                         } else if (key.contains("vendor")) {
753                             if (key.contains("specification")) {
754                                 vendorEvidence.addEvidence(source, key, value, Confidence.LOW);
755                             } else {
756                                 vendorEvidence.addEvidence(source, key, value, Confidence.MEDIUM);
757                                 addMatchingValues(classInformation, value, vendorEvidence);
758                             }
759                         } else if (key.contains("name")) {
760                             productEvidence.addEvidence(source, key, value, Confidence.MEDIUM);
761                             vendorEvidence.addEvidence(source, key, value, Confidence.MEDIUM);
762                             addMatchingValues(classInformation, value, vendorEvidence);
763                             addMatchingValues(classInformation, value, productEvidence);
764                         } else if (key.contains("license")) {
765                             addLicense(dependency, value);
766                         } else {
767                             if (key.contains("description")) {
768                                 addDescription(dependency, value, "manifest", key);
769                             } else {
770                                 productEvidence.addEvidence(source, key, value, Confidence.LOW);
771                                 vendorEvidence.addEvidence(source, key, value, Confidence.LOW);
772                                 addMatchingValues(classInformation, value, vendorEvidence);
773                                 addMatchingValues(classInformation, value, productEvidence);
774                                 if (value.matches(".*\\d.*")) {
775                                     final StringTokenizer tokenizer = new StringTokenizer(value, " ");
776                                     while (tokenizer.hasMoreElements()) {
777                                         final String s = tokenizer.nextToken();
778                                         if (s.matches("^[0-9.]+$")) {
779                                             versionEvidence.addEvidence(source, key, s, Confidence.LOW);
780                                         }
781                                     }
782                                 }
783                             }
784                         }
785                     }
786                 }
787             }
788         } finally {
789             if (jar != null) {
790                 jar.close();
791             }
792         }
793         return foundSomething;
794     }

796     /**
797      * Adds a description to the given dependency. If the description contains one of the following strings beyond 100 characters,
798      * then the description used will be trimmed to that position:
799      * <ul><li>"such as"</li><li>"like "</li><li>"will use "</li><li>"* uses "</li></ul>
800      *
801      * @param dependency a dependency
802      * @param description the description
803      * @param source the source of the evidence
804      * @param key the "name" of the evidence
805      * @return if the description is trimmed, the trimmed version is returned; otherwise the original description is returned
806      */
807     public static String addDescription(Dependency dependency, String description, String source, String key) {
808         if (dependency.getDescription() == null) {
809             dependency.setDescription(description);
810         }
811         String desc;
812         if (HTML_DETECTION_PATTERN.matcher(description).find()) {
813             desc = Jsoup.parse(description).text();
814         } else {
815             desc = description;
816         }
817         dependency.setDescription(desc);
818         if (desc.length() > 100) {
819             desc = desc.replaceAll("\\s\\s+", " ");
820             final int posSuchAs = desc.toLowerCase().indexOf("such as ", 100);
821             final int posLike = desc.toLowerCase().indexOf("like ", 100);
822             final int posWillUse = desc.toLowerCase().indexOf("will use ", 100);
823             final int posUses = desc.toLowerCase().indexOf(" uses ", 100);
824             int pos = -1;
825             pos = Math.max(pos, posSuchAs);
826             if (pos >= 0 && posLike >= 0) {
827                 pos = Math.min(pos, posLike);
828             } else {
829                 pos = Math.max(pos, posLike);
830             }
831             if (pos >= 0 && posWillUse >= 0) {
832                 pos = Math.min(pos, posWillUse);
833             } else {
834                 pos = Math.max(pos, posWillUse);
835             }
836             if (pos >= 0 && posUses >= 0) {
837                 pos = Math.min(pos, posUses);
838             } else {
839                 pos = Math.max(pos, posUses);
840             }
841 
842             if (pos > 0) {
843                 final StringBuilder sb = new StringBuilder(pos + 3);
844                 sb.append(desc.substring(0, pos));
845                 sb.append("...");
846                 desc = sb.toString();
847             }
848             dependency.getProductEvidence().addEvidence(source, key, desc, Confidence.LOW);
849             dependency.getVendorEvidence().addEvidence(source, key, desc, Confidence.LOW);
850         } else {
851             dependency.getProductEvidence().addEvidence(source, key, desc, Confidence.MEDIUM);
852             dependency.getVendorEvidence().addEvidence(source, key, desc, Confidence.MEDIUM);
853         }
854         return desc;
855     }

910     /**
911      * <p>
912      * A utility function that will interpolate strings based on values given in the properties file. It will also interpolate the
913      * strings contained within the properties file so that properties can reference other properties.</p>
914      * <p>
915      * <b>Note:</b> if there is no property found the reference will be removed. In other words, if the interpolated string will
916      * be replaced with an empty string.
917      * </p>
918      * <p>
919      * Example:</p>
920      * <code>
921      * Properties p = new Properties();
922      * p.setProperty("key", "value");
923      * String s = interpolateString("'${key}' and '${nothing}'", p);
924      * System.out.println(s);
925      * </code>
926      * <p>
927      * Will result in:</p>
928      * <code>
929      * 'value' and ''
930      * </code>
931      *
932      * @param text the string that contains references to properties.
933      * @param properties a collection of properties that may be referenced within the text.
934      * @return the interpolated text.
935      */
936     public static String interpolateString(String text, Properties properties) {
937         final Properties props = properties;
938         if (text == null) {
939             return text;
940         }
941         if (props == null) {
942             return text;
943         }
944 
945         final int pos = text.indexOf("${");
946         if (pos < 0) {
947             return text;
948         }
949         final int end = text.indexOf("}");
950         if (end < pos) {
951             return text;
952         }
953 
954         final String propName = text.substring(pos + 2, end);
955         String propValue = interpolateString(props.getProperty(propName), props);
956         if (propValue == null) {
957             propValue = "";
958         }
959         final StringBuilder sb = new StringBuilder(propValue.length() + text.length());
960         sb.append(text.subSequence(0, pos));
961         sb.append(propValue);
962         sb.append(text.substring(end + 1));
963         return interpolateString(sb.toString(), props); //yes yes, this should be a loop...
964     }

966     /**
967      * Determines if the key value pair from the manifest is for an "import" type entry for package names.
968      *
969      * @param key the key from the manifest
970      * @param value the value from the manifest
971      * @return true or false depending on if it is believed the entry is an "import" entry
972      */
973     private boolean isImportPackage(String key, String value) {
974         final Pattern packageRx = Pattern.compile("^([a-zA-Z0-9_#\\$\\*\\.]+\\s*[,;]\\s*)+([a-zA-Z0-9_#\\$\\*\\.]+\\s*)?$");
975         final boolean matches = packageRx.matcher(value).matches();
976         return matches && (key.contains("import") || key.contains("include") || value.length() > 10);
977     }

979     /**
980      * Cycles through an enumeration of JarEntries, contained within the dependency, and returns a list of the class names. This
981      * does not include core Java package names (i.e. java.* or javax.*).
982      *
983      * @param dependency the dependency being analyzed
984      * @return an list of fully qualified class names
985      */
986     private List<ClassNameInformation> collectClassNames(Dependency dependency) {
987         final List<ClassNameInformation> classNames = new ArrayList<ClassNameInformation>();
988         JarFile jar = null;
989         try {
990             jar = new JarFile(dependency.getActualFilePath());
991             final Enumeration entries = jar.entries();
992             while (entries.hasMoreElements()) {
993                 final JarEntry entry = (JarEntry) entries.nextElement();
994                 final String name = entry.getName().toLowerCase();
995                 //no longer stripping "|com\\.sun" - there are some com.sun jar files with CVEs.
996                 if (name.endsWith(".class") && !name.matches("^javax?\\..*$")) {
997                     final ClassNameInformation className = new ClassNameInformation(name.substring(0, name.length() - 6));
998                     classNames.add(className);
999                 }
1000             }
1001         } catch (IOException ex) {
1002             final String msg = String.format("Unable to open jar file '%s'.", dependency.getFileName());
1003             LOGGER.log(Level.WARNING, msg);
1004             LOGGER.log(Level.FINE, null, ex);
1005         } finally {
1006             if (jar != null) {
1007                 try {
1008                     jar.close();
1009                 } catch (IOException ex) {
1010                     LOGGER.log(Level.FINEST, null, ex);
1011                 }
1012             }
1013         }
1014         return classNames;
1015     }

1017     /**
1018      * Cycles through the list of class names and places the package levels 0-3 into the provided maps for vendor and product.
1019      * This is helpful when analyzing vendor/product as many times this is included in the package name.
1020      *
1021      * @param classNames a list of class names
1022      * @param vendor HashMap of possible vendor names from package names (e.g. owasp)
1023      * @param product HashMap of possible product names from package names (e.g. dependencycheck)
1024      */
1025     private void analyzeFullyQualifiedClassNames(List<ClassNameInformation> classNames,
1026             Map<String, Integer> vendor, Map<String, Integer> product) {
1027         for (ClassNameInformation entry : classNames) {
1028             final List<String> list = entry.getPackageStructure();
1029             addEntry(vendor, list.get(0));
1030 
1031             if (list.size() == 2) {
1032                 addEntry(product, list.get(1));
1033             }
1034             if (list.size() == 3) {
1035                 addEntry(vendor, list.get(1));
1036                 addEntry(product, list.get(1));
1037                 addEntry(product, list.get(2));
1038             }
1039             if (list.size() >= 4) {
1040                 addEntry(vendor, list.get(1));
1041                 addEntry(vendor, list.get(2));
1042                 addEntry(product, list.get(1));
1043                 addEntry(product, list.get(2));
1044                 addEntry(product, list.get(3));
1045             }
1046         }

1065      * Cycles through the collection of class name information to see if parts of the package names are contained in the provided
1066      * value. If found, it will be added as the HIGHEST confidence evidence because we have more then one source corroborating the
1067      * value.

1142     /**
1143      * Stores information about a class name.
1144      */
1145     protected static class ClassNameInformation {
1146 
1147         /**
1148          * <p>
1149          * Stores information about a given class name. This class will keep the fully qualified class name and a list of the
1150          * important parts of the package structure. Up to the first four levels of the package structure are stored, excluding a
1151          * leading "org" or "com". Example:</p>
1152          * <code>ClassNameInformation obj = new ClassNameInformation("org.owasp.dependencycheck.analyzer.JarAnalyzer");
1153          * System.out.println(obj.getName());
1154          * for (String p : obj.getPackageStructure())
1155          *     System.out.println(p);
1156          * </code>
1157          * <p>
1158          * Would result in:</p>
1159          * <code>org.owasp.dependencycheck.analyzer.JarAnalyzer
1160          * owasp
1161          * dependencycheck
1162          * analyzer
1163          * jaranalyzer</code>
1164          *
1165          * @param className a fully qualified class name
1166          */
1167         ClassNameInformation(String className) {
1168             name = className;
1169             if (name.contains("/")) {
1170                 final String[] tmp = className.toLowerCase().split("/");
1171                 int start = 0;
1172                 int end = 3;
1173                 if ("com".equals(tmp[0]) || "org".equals(tmp[0])) {
1174                     start = 1;
1175                     end = 4;
1176                 }
1177                 if (tmp.length <= end) {
1178                     end = tmp.length - 1;
1179                 }
1180                 for (int i = start; i <= end; i++) {
1181                     packageStructure.add(tmp[i]);
1182                 }
1183             } else {
1184                 packageStructure.add(name);
1185             }
1186         }
1187         /**
1188          * The fully qualified class name.
1189          */
1190         private String name;

1193          * Get the value of name
1194          *

1203          *
1204          * @param name new value of name
1205          */
1206         public void setName(String name) {

1208         }

1210          * Up to the first four levels of the package structure, excluding a leading "org" or "com".
1211          */
1212         private final ArrayList<String> packageStructure = new ArrayList<String>();
